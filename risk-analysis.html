Overview

This section performs comprehensive interest rate risk analysis using the Dothan model and simulated bond prices. We execute four key tasks:

  1. Impact of Interest Rate Volatility: Analyze how bond prices change with different volatility levels

  2. Initial Interest Rate Changes: Assess sensitivity to starting rate level

  3. Bond Portfolio Risk: Evaluate risk in a realistic portfolio

  4. Interest Rate Shock Scenario: Stress test under sudden rate increases

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from scipy import stats

class DothanSimulator:
    """Efficient Dothan model interest rate path simulator"""
    
    def __init__(self, r0, mu, sigma, T, dt=0.01):
        """
        Initialize the simulator
        
        Parameters
        ----------
        r0 : float
            Initial short rate (e.g., 0.03 for 3%)
        mu : float
            Drift parameter (annual, e.g., 0.02 for 2%)
        sigma : float
            Volatility parameter (annual, e.g., 0.15 for 15%)
        T : float
            Time horizon in years (e.g., 10)
        dt : float
            Time step in years (e.g., 0.01 for ~2.5 trading days)
        """
        self.r0 = r0
        self.mu = mu
        self.sigma = sigma
        self.T = T
        self.dt = dt
        self.n_steps = int(T / dt)
        self.time_grid = np.linspace(0, T, self.n_steps)
        
        # Precompute constants for efficiency
        self.drift_coeff = self.mu * self.dt
        self.diffusion_coeff = self.sigma * np.sqrt(self.dt)
        
    def simulate_paths(self, n_paths, seed=None):
        """
        Simulate multiple interest rate paths efficiently
        
        Parameters
        ----------
        n_paths : int
            Number of paths to simulate (e.g., 10000)
        seed : int, optional
            Random seed for reproducibility
        
        Returns
        -------
        paths : ndarray of shape (n_steps, n_paths)
            Simulated interest rate paths
        """
        if seed is not None:
            np.random.seed(seed)
        
        # Initialize array
        paths = np.zeros((self.n_steps, n_paths))
        paths[0, :] = self.r0
        
        # Generate all random variables at once (vectorized)
        Z = np.random.standard_normal((self.n_steps - 1, n_paths))
        
        # Simulation loop (vectorized over paths)
        for t in range(1, self.n_steps):
            drift = self.drift_coeff * paths[t-1, :]
            diffusion = self.diffusion_coeff * paths[t-1, :] * Z[t-1, :]
            paths[t, :] = paths[t-1, :] + drift + diffusion
            
            # Optional: enforce positive rates
            paths[t, :] = np.maximum(paths[t, :], 1e-8)
        
        return paths
    
    def statistics(self, paths):
        """Compute comprehensive path statistics"""
        stats_dict = {
            'mean_path': np.mean(paths, axis=1),
            'median_path': np.median(paths, axis=1),
            'std_path': np.std(paths, axis=1),
            'min_path': np.min(paths, axis=1),
            'max_path': np.max(paths, axis=1),
            'percentile_05': np.percentile(paths, 5, axis=1),
            'percentile_95': np.percentile(paths, 95, axis=1),
            'percentile_25': np.percentile(paths, 25, axis=1),
            'percentile_75': np.percentile(paths, 75, axis=1),
        }
        
        # Final statistics
        final_rates = paths[-1, :]
        stats_dict['final_mean'] = np.mean(final_rates)
        stats_dict['final_median'] = np.median(final_rates)
        stats_dict['final_std'] = np.std(final_rates)
        stats_dict['final_min'] = np.min(final_rates)
        stats_dict['final_max'] = np.max(final_rates)
        
        return stats_dict

class BondPricingEngine:
    """Price zero-coupon bonds using simulated interest rate paths"""
    
    def __init__(self, dt=0.01):
        """
        Parameters
        ----------
        dt : float
            Time step used in path simulation (must match simulator)
        """
        self.dt = dt
    
    def price_from_paths(self, paths, maturity):
        """
        Price a zero-coupon bond maturing at 'maturity' 
        using simulated interest rate paths
        
        Parameters
        ----------
        paths : ndarray of shape (n_steps, n_paths)
            Simulated interest rate paths
        maturity : float
            Bond maturity in years
        
        Returns
        -------
        bond_price : float
            Estimated bond price
        bond_std : float
            Standard error of estimate
        bond_ci : tuple
            95% confidence interval (lower, upper)
        """
        n_steps = paths.shape[0]
        n_paths = paths.shape[1]
        
        # Convert maturity to step index
        maturity_step = int(maturity / self.dt)
        
        # Ensure maturity is within simulation horizon
        if maturity_step > n_steps:
            raise ValueError(f"Maturity {maturity} exceeds simulation horizon")
        
        # Accumulate discount factor for each path
        discount_factors = np.zeros(n_paths)
        
        for i in range(n_paths):
            # Sum of interest rates up to maturity
            rate_integral = np.sum(paths[:maturity_step, i]) * self.dt
            # Discount factor
            discount_factors[i] = np.exp(-rate_integral)
        
        # Compute statistics
        bond_price = np.mean(discount_factors)
        bond_std = np.std(discount_factors) / np.sqrt(n_paths)
        z_critical = 1.96  # 95% confidence
        ci_lower = bond_price - z_critical * bond_std
        ci_upper = bond_price + z_critical * bond_std
        
        return {
            'price': bond_price,
            'std_error': bond_std,
            'ci_lower': ci_lower,
            'ci_upper': ci_upper,
            'relative_error': bond_std / bond_price * 100
        }

print("Simulator and engine initialized.")

# Create pricer instance
pricer = BondPricingEngine(dt=0.01)

TASK 1: Impact of Interest Rate Volatility

Objective

Understand how uncertainty in future interest rates affects bond prices. Higher volatility increases uncertainty, which (paradoxically) increases bond prices due to convexity.

Analysis

portfolio_bonds = {
    'Bond A (2Y)': (2, 0.30),

total_portfolio_value = 0
for bond_name, (maturity, weight) in portfolio_bonds.items():
    result = pricer.price_from_paths(paths_baseline, maturity)
    bond_price = result['price']
    
    # Assume $1M total portfolio
    position_weight = weight
    position_value = 1_000_000 * weight
    position_units = position_value / bond_price
    
    portfolio_prices[bond_name] = bond_price
    bond_details[bond_name] = {
        'maturity': maturity,
        'weight': weight,
        'price': bond_price,
        'position_value': position_value,
        'units': position_units
    }
    total_portfolio_value += position_value
    
    print(f"{bond_name:17} | {maturity:8.0f} | {weight:5.1%} | ${bond_price:13.4f} | "
          f"${position_value:>12,.0f}")

print("-" * 100)
print(f"{'Total Portfolio Value':36} | ${total_portfolio_value:>12,.0f}")
print("-" * 100)

# Compute portfolio metrics
print("\nPortfolio Risk Metrics:")
print("-" * 100)

# Duration (weighted average)
durations = (2, 5, 10)
weights = [bd['weight'] for bd in bond_details.values()]
portfolio_duration = sum(d * w for d, w in zip(durations, weights))

print(f"Portfolio Duration:        {portfolio_duration:.2f} years")
print(f"Portfolio Value:           ${total_portfolio_value:,.0f}")

# Compute value changes under different scenarios
scenarios = {
    'Δr = +0.5%': 0.005,
    'Δr = +1.0%': 0.010,
    'Δr = +1.5%': 0.015,
    'Δr = +2.0%': 0.020,
}

scenario_results = []

for scenario_name, rate_shock in scenarios.items():
    # Simulate with shifted initial rate
    sim_shocked = DothanSimulator(r0=0.03 + rate_shock, mu=0.02, sigma=0.15, T=10, dt=0.01)
    paths_shocked = sim_shocked.simulate_paths(n_paths=10000, seed=42)
    
    portfolio_value_shocked = 0
    
    for bond_name, (maturity, weight) in portfolio_bonds.items():
        result = pricer.price_from_paths(paths_shocked, maturity)
        new_price = result['price']
        
        # New position value
        units = bond_details[bond_name]['units']
        new_position_value = units * new_price
        portfolio_value_shocked += new_position_value
    
    portfolio_loss = portfolio_value_shocked - total_portfolio_value
    portfolio_loss_pct = portfolio_loss / total_portfolio_value * 100
    
    scenario_results.append({
        'Scenario': scenario_name,
        'New Portfolio Value': portfolio_value_shocked,
        'Loss ($)': portfolio_loss,
        'Loss (%)': portfolio_loss_pct
    })
    
    print(f"\n{scenario_name}:")
    print(f"  New portfolio value: ${portfolio_value_shocked:>12,.0f}")
    print(f"  Portfolio loss:      ${portfolio_loss:>12,.0f} ({portfolio_loss_pct:>6.2f}%)")

#| label: setup-risk-analysis
#| code-fold: true

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from scipy import stats

class DothanSimulator:
    """Efficient Dothan model interest rate path simulator"""
    
    def __init__(self, r0, mu, sigma, T, dt=0.01):
        """
        Initialize the simulator
        
        Parameters
        ----------
        r0 : float
            Initial short rate (e.g., 0.03 for 3%)
        mu : float
            Drift parameter (annual, e.g., 0.02 for 2%)
        sigma : float
            Volatility parameter (annual, e.g., 0.15 for 15%)
        T : float
            Time horizon in years (e.g., 10)
        dt : float
            Time step in years (e.g., 0.01 for ~2.5 trading days)
        """
        self.r0 = r0
        self.mu = mu
        self.sigma = sigma
        self.T = T
        self.dt = dt
        self.n_steps = int(T / dt)
        self.time_grid = np.linspace(0, T, self.n_steps)
        
        # Precompute constants for efficiency
        self.drift_coeff = self.mu * self.dt
        self.diffusion_coeff = self.sigma * np.sqrt(self.dt)
        
    def simulate_paths(self, n_paths, seed=None):
        """
        Simulate multiple interest rate paths efficiently
        
        Parameters
        ----------
        n_paths : int
            Number of paths to simulate (e.g., 10000)
        seed : int, optional
            Random seed for reproducibility
        
        Returns
        -------
        paths : ndarray of shape (n_steps, n_paths)
            Simulated interest rate paths
        """
        if seed is not None:
            np.random.seed(seed)
        
        # Initialize array
        paths = np.zeros((self.n_steps, n_paths))
        paths[0, :] = self.r0
        
        # Generate all random variables at once (vectorized)
        Z = np.random.standard_normal((self.n_steps - 1, n_paths))
        
        # Simulation loop (vectorized over paths)
        for t in range(1, self.n_steps):
            drift = self.drift_coeff * paths[t-1, :]
            diffusion = self.diffusion_coeff * paths[t-1, :] * Z[t-1, :]
            paths[t, :] = paths[t-1, :] + drift + diffusion
            
            # Optional: enforce positive rates
            paths[t, :] = np.maximum(paths[t, :], 1e-8)
        
        return paths
    
    def statistics(self, paths):
        """Compute comprehensive path statistics"""
        stats_dict = {
            'mean_path': np.mean(paths, axis=1),
            'median_path': np.median(paths, axis=1),
            'std_path': np.std(paths, axis=1),
            'min_path': np.min(paths, axis=1),
            'max_path': np.max(paths, axis=1),
            'percentile_05': np.percentile(paths, 5, axis=1),
            'percentile_95': np.percentile(paths, 95, axis=1),
            'percentile_25': np.percentile(paths, 25, axis=1),
            'percentile_75': np.percentile(paths, 75, axis=1),
        }
        
        # Final statistics
        final_rates = paths[-1, :]
        stats_dict['final_mean'] = np.mean(final_rates)
        stats_dict['final_median'] = np.median(final_rates)
        stats_dict['final_std'] = np.std(final_rates)
        stats_dict['final_min'] = np.min(final_rates)
        stats_dict['final_max'] = np.max(final_rates)
        
        return stats_dict

class BondPricingEngine:
    """Price zero-coupon bonds using simulated interest rate paths"""
    
    def __init__(self, dt=0.01):
        """
        Parameters
        ----------
        dt : float
            Time step used in path simulation (must match simulator)
        """
        self.dt = dt
    
    def price_from_paths(self, paths, maturity):
        """
        Price a zero-coupon bond maturing at 'maturity' 
        using simulated interest rate paths
        
        Parameters
        ----------
        paths : ndarray of shape (n_steps, n_paths)
            Simulated interest rate paths
        maturity : float
            Bond maturity in years
        
        Returns
        -------
        bond_price : float
            Estimated bond price
        bond_std : float
            Standard error of estimate
        bond_ci : tuple
            95% confidence interval (lower, upper)
        """
        n_steps = paths.shape[0]
        n_paths = paths.shape[1]
        
        # Convert maturity to step index
        maturity_step = int(maturity / self.dt)
        
        # Ensure maturity is within simulation horizon
        if maturity_step > n_steps:
            raise ValueError(f"Maturity {maturity} exceeds simulation horizon")
        
        # Accumulate discount factor for each path
        discount_factors = np.zeros(n_paths)
        
        for i in range(n_paths):
            # Sum of interest rates up to maturity
            rate_integral = np.sum(paths[:maturity_step, i]) * self.dt
            # Discount factor
            discount_factors[i] = np.exp(-rate_integral)
        
        # Compute statistics
        bond_price = np.mean(discount_factors)
        bond_std = np.std(discount_factors) / np.sqrt(n_paths)
        z_critical = 1.96  # 95% confidence
        ci_lower = bond_price - z_critical * bond_std
        ci_upper = bond_price + z_critical * bond_std
        
        return {
            'price': bond_price,
            'std_error': bond_std,
            'ci_lower': ci_lower,
            'ci_upper': ci_upper,
            'relative_error': bond_std / bond_price * 100
        }

print("Simulator and engine initialized.")

# Create pricer instance
pricer = BondPricingEngine(dt=0.01)

## Risk Management Implications

1. **Duration is Critical**: Portfolio duration directly determines interest rate sensitivity
2. **Convexity Matters**: Non-linear effects mean duration alone is insufficient for large shocks
3. **Concentration Risk**: Long-duration bonds amplify losses in adverse scenarios
4. **Hedging Strategies**:
   - Sell long-duration bonds, buy short-duration (barbell strategy)
   - Use interest rate swaps to reduce duration exposure
   - Diversify across maturities to reduce concentration risk
