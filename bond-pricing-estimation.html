Objective

Using the simulated interest rate paths from the Dothan model, we now estimate zero-coupon bond prices for multiple maturities. This section ties together simulation and bond pricing.

Methodology

Bond Pricing Algorithm Recap

For each simulated path \(i = 1, 2, ..., M\):

  1. Accumulate interest: \(I_i = \sum_{j=1}^{n_T} r_{t_j} \Delta t\) where \(n_T = T / \Delta t\)

  2. Compute discount factor: \(D_i = \exp(-I_i)\)

  3. Average across paths: \[P(0,T) = \frac{1}{M} \sum_{i=1}^M D_i\]

Statistical Measures

For accurate estimation, we compute:

Metric Formula Interpretation
Point Estimate \(\\bar{P} = \\frac{1}{M} \\sum_i D_i\) Best estimate of bond price
Standard Error \(SE = \\frac{\\sigma(D_i)}{\\sqrt{M}}\) Estimation uncertainty
95% Confidence Interval \([\\bar{P} \\pm 1.96 \\cdot SE]\) Range containing true price (95% confidence)
Relative Error \(\\frac{SE}{\\bar{P}} \\times 100\\%\) Percentage estimation uncertainty

Where \(\\sigma(D_i)\) is the sample standard deviation of discount factors.

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from scipy import stats

class DothanSimulator:
    """Efficient Dothan model interest rate path simulator"""
    
    def __init__(self, r0, mu, sigma, T, dt=0.01):
        """
        Initialize the simulator
        
        Parameters
        ----------
        r0 : float
            Initial short rate (e.g., 0.03 for 3%)
        mu : float
            Drift parameter (annual, e.g., 0.02 for 2%)
        sigma : float
            Volatility parameter (annual, e.g., 0.15 for 15%)
        T : float
            Time horizon in years (e.g., 10)
        dt : float
            Time step in years (e.g., 0.01 for ~2.5 trading days)
        """
        self.r0 = r0
        self.mu = mu
        self.sigma = sigma
        self.T = T
        self.dt = dt
        self.n_steps = int(T / dt)
        self.time_grid = np.linspace(0, T, self.n_steps)
        
        # Precompute constants for efficiency
        self.drift_coeff = self.mu * self.dt
        self.diffusion_coeff = self.sigma * np.sqrt(self.dt)
        
    def simulate_paths(self, n_paths, seed=None):
        """
        Simulate multiple interest rate paths efficiently
        
        Parameters
        ----------
        n_paths : int
            Number of paths to simulate (e.g., 10000)
        seed : int, optional
            Random seed for reproducibility
        
        Returns
        -------
        paths : ndarray of shape (n_steps, n_paths)
            Simulated interest rate paths
        """
        if seed is not None:
            np.random.seed(seed)
        
        # Initialize array
        paths = np.zeros((self.n_steps, n_paths))
        paths[0, :] = self.r0
        
        # Generate all random variables at once (vectorized)
        Z = np.random.standard_normal((self.n_steps - 1, n_paths))
        
        # Simulation loop (vectorized over paths)
        for t in range(1, self.n_steps):
            drift = self.drift_coeff * paths[t-1, :]
            diffusion = self.diffusion_coeff * paths[t-1, :] * Z[t-1, :]
            paths[t, :] = paths[t-1, :] + drift + diffusion
            
            # Optional: enforce positive rates
            paths[t, :] = np.maximum(paths[t, :], 1e-8)
        
        return paths
    
    def statistics(self, paths):
        """Compute comprehensive path statistics"""
        stats_dict = {
            'mean_path': np.mean(paths, axis=1),
            'median_path': np.median(paths, axis=1),
            'std_path': np.std(paths, axis=1),
            'min_path': np.min(paths, axis=1),
            'max_path': np.max(paths, axis=1),
            'percentile_05': np.percentile(paths, 5, axis=1),
            'percentile_95': np.percentile(paths, 95, axis=1),
            'percentile_25': np.percentile(paths, 25, axis=1),
            'percentile_75': np.percentile(paths, 75, axis=1),
        }
        
        # Final statistics
        final_rates = paths[-1, :]
        stats_dict['final_mean'] = np.mean(final_rates)
        stats_dict['final_median'] = np.median(final_rates)
        stats_dict['final_std'] = np.std(final_rates)
        stats_dict['final_min'] = np.min(final_rates)
        stats_dict['final_max'] = np.max(final_rates)
        
        return stats_dict

Complete Bond Pricing Implementation

class BondPricingEngine:
    """Price zero-coupon bonds using simulated interest rate paths"""
    
    def __init__(self, dt=0.01):
        """
        Parameters
        ----------
        dt : float
            Time step used in path simulation (must match simulator)
        """
        self.dt = dt
    
    def price_from_paths(self, paths, maturity):
        """
        Price a zero-coupon bond maturing at 'maturity' 
        using simulated interest rate paths
        
        Parameters
        ----------
        paths : ndarray of shape (n_steps, n_paths)
            Simulated interest rate paths
        maturity : float
            Bond maturity in years
        
        Returns
        -------
        bond_price : float
            Estimated bond price
        bond_std : float
            Standard error of estimate
        bond_ci : tuple
            95% confidence interval (lower, upper)
        """
        n_steps = paths.shape[0]
        n_paths = paths.shape[1]
        
        # Convert maturity to step index
        maturity_step = int(maturity / self.dt)
        
        # Ensure maturity is within simulation horizon
        if maturity_step > n_steps:
            raise ValueError(f"Maturity {maturity} exceeds simulation horizon")
        
        # Accumulate discount factor for each path
        discount_factors = np.zeros(n_paths)
        
        for i in range(n_paths):
            # Sum of interest rates up to maturity
            rate_integral = np.sum(paths[:maturity_step, i]) * self.dt
            # Discount factor
            discount_factors[i] = np.exp(-rate_integral)
        
        # Compute statistics
        bond_price = np.mean(discount_factors)
        bond_std = np.std(discount_factors) / np.sqrt(n_paths)
        z_critical = 1.96  # 95% confidence
        ci_lower = bond_price - z_critical * bond_std
        ci_upper = bond_price + z_critical * bond_std
        
        return {
            'price': bond_price,
            'std_error': bond_std,
            'ci_lower': ci_lower,
            'ci_upper': ci_upper,
            'relative_error': bond_std / bond_price * 100
        }
    
    def price_portfolio(self, paths, bonds_dict):
        """
        Price a portfolio of bonds
        
        Parameters
        ----------
        paths : ndarray
            Simulated interest rate paths
        bonds_dict : dict
            Dictionary of {name: (maturity, weight)}
            Example: {'Bond_A': (2, 0.3), 'Bond_B': (5, 0.4), 'Bond_C': (10, 0.3)}
        
        Returns
        -------
        portfolio_results : dict
            Results for each bond and portfolio
        """
        results = {}
        portfolio_value = 0
        total_weight = 0
        
        for bond_name, (maturity, weight) in bonds_dict.items():
            bond_result = self.price_from_paths(paths, maturity)
            bond_result['weight'] = weight
            bond_result['weighted_price'] = bond_result['price'] * weight
            results[bond_name] = bond_result
            
            portfolio_value += bond_result['weighted_price']
            total_weight += weight
        
        # Portfolio results
        results['Portfolio'] = {
            'price': portfolio_value / total_weight if total_weight > 0 else 0,
            'weights': total_weight
        }
        
        return results
    
    def price_across_maturities(self, paths, maturities):
        """
        Price bonds for multiple maturities
        
        Parameters
        ----------
        paths : ndarray
            Simulated paths
        maturities : array-like
            Array of maturities to price
        
        Returns
        -------
        price_table : pd.DataFrame
            Table with prices, errors, and CI
        """
        results = []
        
        for T in maturities:
            result = self.price_from_paths(paths, T)
            results.append({
                'Maturity (years)': T,
                'Price': result['price'],
                'Std Error': result['std_error'],
                'CI Lower': result['ci_lower'],
                'CI Upper': result['ci_upper'],
                'Rel Error (%)': result['relative_error']
            })
        
        return pd.DataFrame(results)

## Objective

Using the simulated interest rate paths from the Dothan model, we now estimate zero-coupon bond prices for multiple maturities. This section ties together simulation and bond pricing.

## Methodology

### Bond Pricing Algorithm Recap

For each simulated path $i = 1, 2, ..., M$:

1. **Accumulate interest**: $I_i = \sum_{j=1}^{n_T} r_{t_j} \Delta t$ where $n_T = T / \Delta t$

2. **Compute discount factor**: $D_i = \exp(-I_i)$

3. **Average across paths**: $$P(0,T) = \frac{1}{M} \sum_{i=1}^M D_i$$

### Statistical Measures

For accurate estimation, we compute:

| Metric | Formula | Interpretation |
|--------|---------|-----------------|
| **Point Estimate** | $\\bar{P} = \\frac{1}{M} \\sum_i D_i$ | Best estimate of bond price |
| **Standard Error** | $SE = \\frac{\\sigma(D_i)}{\\sqrt{M}}$ | Estimation uncertainty |
| **95% Confidence Interval** | $[\\bar{P} \\pm 1.96 \\cdot SE]$ | Range containing true price (95% confidence) |
| **Relative Error** | $\\frac{SE}{\\bar{P}} \\times 100\\%$ | Percentage estimation uncertainty |

Where $\\sigma(D_i)$ is the sample standard deviation of discount factors.

## Setup: Generate Interest Rate Paths

First, we recreate the interest rate paths from the Dothan model:

```{python}
#| label: setup-paths
#| code-fold: true

import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from scipy import stats

class DothanSimulator:
    """Efficient Dothan model interest rate path simulator"""
    
    def __init__(self, r0, mu, sigma, T, dt=0.01):
        """
        Initialize the simulator
        
        Parameters
        ----------
        r0 : float
            Initial short rate (e.g., 0.03 for 3%)
        mu : float
            Drift parameter (annual, e.g., 0.02 for 2%)
        sigma : float
            Volatility parameter (annual, e.g., 0.15 for 15%)
        T : float
            Time horizon in years (e.g., 10)
        dt : float
            Time step in years (e.g., 0.01 for ~2.5 trading days)
        """
        self.r0 = r0
        self.mu = mu
        self.sigma = sigma
        self.T = T
        self.dt = dt
        self.n_steps = int(T / dt)
        self.time_grid = np.linspace(0, T, self.n_steps)
        
        # Precompute constants for efficiency
        self.drift_coeff = self.mu * self.dt
        self.diffusion_coeff = self.sigma * np.sqrt(self.dt)
        
    def simulate_paths(self, n_paths, seed=None):
        """
        Simulate multiple interest rate paths efficiently
        
        Parameters
        ----------
        n_paths : int
            Number of paths to simulate (e.g., 10000)
        seed : int, optional
            Random seed for reproducibility
        
        Returns
        -------
        paths : ndarray of shape (n_steps, n_paths)
            Simulated interest rate paths
        """
        if seed is not None:
            np.random.seed(seed)
        
        # Initialize array
        paths = np.zeros((self.n_steps, n_paths))
        paths[0, :] = self.r0
        
        # Generate all random variables at once (vectorized)
        Z = np.random.standard_normal((self.n_steps - 1, n_paths))
        
        # Simulation loop (vectorized over paths)
        for t in range(1, self.n_steps):
            drift = self.drift_coeff * paths[t-1, :]
            diffusion = self.diffusion_coeff * paths[t-1, :] * Z[t-1, :]
            paths[t, :] = paths[t-1, :] + drift + diffusion
            
            # Optional: enforce positive rates
            paths[t, :] = np.maximum(paths[t, :], 1e-8)
        
        return paths
    
    def statistics(self, paths):
        """Compute comprehensive path statistics"""
        stats_dict = {
            'mean_path': np.mean(paths, axis=1),
            'median_path': np.median(paths, axis=1),
            'std_path': np.std(paths, axis=1),
            'min_path': np.min(paths, axis=1),
            'max_path': np.max(paths, axis=1),
            'percentile_05': np.percentile(paths, 5, axis=1),
            'percentile_95': np.percentile(paths, 95, axis=1),
            'percentile_25': np.percentile(paths, 25, axis=1),
            'percentile_75': np.percentile(paths, 75, axis=1),
        }
        
        # Final statistics
        final_rates = paths[-1, :]
        stats_dict['final_mean'] = np.mean(final_rates)
        stats_dict['final_median'] = np.median(final_rates)
        stats_dict['final_std'] = np.std(final_rates)
        stats_dict['final_min'] = np.min(final_rates)
        stats_dict['final_max'] = np.max(final_rates)
        
        return stats_dict

# Run simulation
simulator = DothanSimulator(r0=0.03, mu=0.02, sigma=0.15, T=10, dt=0.01)
paths = simulator.simulate_paths(n_paths=5000, seed=42)
stats_result = simulator.statistics(paths)

print("Simulation completed. 5000 interest rate paths generated.")
```

```{python}
#| label: bond-pricing-complete
#| code-fold: false

class BondPricingEngine:
    """Price zero-coupon bonds using simulated interest rate paths"""
    
    def __init__(self, dt=0.01):
        """
        Parameters
        ----------
        dt : float
            Time step used in path simulation (must match simulator)
        """
        self.dt = dt
    
    def price_from_paths(self, paths, maturity):
        """
        Price a zero-coupon bond maturing at 'maturity' 
        using simulated interest rate paths
        
        Parameters
        ----------
        paths : ndarray of shape (n_steps, n_paths)
            Simulated interest rate paths
        maturity : float
            Bond maturity in years
        
        Returns
        -------
        bond_price : float
            Estimated bond price
        bond_std : float
            Standard error of estimate
        bond_ci : tuple
            95% confidence interval (lower, upper)
        """
        n_steps = paths.shape[0]
        n_paths = paths.shape[1]
        
        # Convert maturity to step index
        maturity_step = int(maturity / self.dt)
        
        # Ensure maturity is within simulation horizon
        if maturity_step > n_steps:
            raise ValueError(f"Maturity {maturity} exceeds simulation horizon")
        
        # Accumulate discount factor for each path
        discount_factors = np.zeros(n_paths)
        
        for i in range(n_paths):
            # Sum of interest rates up to maturity
            rate_integral = np.sum(paths[:maturity_step, i]) * self.dt
            # Discount factor
            discount_factors[i] = np.exp(-rate_integral)
        
        # Compute statistics
        bond_price = np.mean(discount_factors)
        bond_std = np.std(discount_factors) / np.sqrt(n_paths)
        z_critical = 1.96  # 95% confidence
        ci_lower = bond_price - z_critical * bond_std
        ci_upper = bond_price + z_critical * bond_std
        
        return {
            'price': bond_price,
            'std_error': bond_std,
            'ci_lower': ci_lower,
            'ci_upper': ci_upper,
            'relative_error': bond_std / bond_price * 100
        }
    
    def price_portfolio(self, paths, bonds_dict):
        """
        Price a portfolio of bonds
        
        Parameters
        ----------
        paths : ndarray
            Simulated interest rate paths
        bonds_dict : dict
            Dictionary of {name: (maturity, weight)}
            Example: {'Bond_A': (2, 0.3), 'Bond_B': (5, 0.4), 'Bond_C': (10, 0.3)}
        
        Returns
        -------
        portfolio_results : dict
            Results for each bond and portfolio
        """
        results = {}
        portfolio_value = 0
        total_weight = 0
        
        for bond_name, (maturity, weight) in bonds_dict.items():
            bond_result = self.price_from_paths(paths, maturity)
            bond_result['weight'] = weight
            bond_result['weighted_price'] = bond_result['price'] * weight
            results[bond_name] = bond_result
            
            portfolio_value += bond_result['weighted_price']
            total_weight += weight
        
        # Portfolio results
        results['Portfolio'] = {
            'price': portfolio_value / total_weight if total_weight > 0 else 0,
            'weights': total_weight
        }
        
        return results
    
    def price_across_maturities(self, paths, maturities):
        """
        Price bonds for multiple maturities
        
        Parameters
        ----------
        paths : ndarray
            Simulated paths
        maturities : array-like
            Array of maturities to price
        
        Returns
        -------
        price_table : pd.DataFrame
            Table with prices, errors, and CI
        """
        results = []
        
        for T in maturities:
            result = self.price_from_paths(paths, T)
            results.append({
                'Maturity (years)': T,
                'Price': result['price'],
                'Std Error': result['std_error'],
                'CI Lower': result['ci_lower'],
                'CI Upper': result['ci_upper'],
                'Rel Error (%)': result['relative_error']
            })
        
        return pd.DataFrame(results)

# ============================================================================
```
